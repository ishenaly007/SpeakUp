    <h2>The Poet's Toolkit</h2>
    <p>Poets use various literary devices to enhance their work:</p>
    <ul>
      <li><strong>Metaphor:</strong> Comparing two unlike things without using "like" or "as". <em>(e.g., "The world is a stage.")</em></li>
      <li><strong>Simile:</strong> Comparing two unlike things using "like" or "as". <em>(e.g., "She floated in like a cloud.")</em></li>
      <li><strong>Imagery:</strong> Language that appeals to the senses.</li>
      <li><strong>Alliteration:</strong> Repetition of initial consonant sounds. <em>(e.g., "Peter Piper picked...")</em></li>
      <li><strong>Rhyme & Rhythm:</strong> The musical qualities of language.</li>
    </ul>
  </section>

document.addEventListener('DOMContentLoaded', () => {
  const poetryContainer = document.getElementById('lesson-poetry-in-motion');
  if (!poetryContainer) return;

  const haikuInputs = poetryContainer.querySelectorAll('.haiku-writer input[type="text"]');

  haikuInputs.forEach(input => {
    const syllableTarget = parseInt(input.dataset.syllables, 10);
    const countDisplay = input.parentElement.querySelector('.syllable-count');

    input.addEventListener('input', () => {
      // Basic syllable approximation (counts groups of vowels, very rough)
      // This is a placeholder for a more complex syllable counter.
      // For this exercise, we'll do a simpler character count as a proxy or word count.
      // Let's use word count for simplicity here as true syllable counting is complex.
      // Or, let's just count characters as a simple proxy.
      const text = input.value.trim();
      let currentSyllables = 0;

      if (text.length > 0) {
        // Very naive "syllable" count based on non-space characters / average word length.
        // This is NOT accurate for real syllables.
        // For a slightly better naive approach, count vowel groups.
        const vowelGroups = text.match(/[aeiouyAEIOUY]+/g);
        currentSyllables = vowelGroups ? vowelGroups.length : 0;
        
        // A very simple adjustment for short words / single vowels
        if (text.length <= 3 && currentSyllables === 0 && text.length > 0) {
            currentSyllables = 1;
        }
      }
      
      if (countDisplay) {
        countDisplay.textContent = `${currentSyllables}/${syllableTarget}`;
        if (currentSyllables > syllableTarget) {
          countDisplay.style.color = 'red';
        } else if (currentSyllables === syllableTarget) {
          countDisplay.style.color = 'green';
        } else {
          countDisplay.style.color = '#777';
        }
      }
    });
  });
});
