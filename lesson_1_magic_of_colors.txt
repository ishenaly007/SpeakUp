JavaScript Content:
document.addEventListener('DOMContentLoaded', () => {
  const lessonContainer = document.querySelector('#lesson-magic-of-colors');
  if (!lessonContainer) return;

  const displayBox = lessonContainer.querySelector('#interactive-color-display');
  const buttons = lessonContainer.querySelectorAll('#color-interaction button');

  buttons.forEach(button => {
    button.addEventListener('click', () => {
      const color = button.dataset.color;
      if (displayBox) {
        displayBox.style.backgroundColor = color;
        displayBox.textContent = `Selected: ${color.charAt(0).toUpperCase() + color.slice(1)}`;
        if (color === 'yellow') {
            displayBox.style.color = '#333';
        } else {
            displayBox.style.color = 'white';
        }
      }
    });
  });
});
